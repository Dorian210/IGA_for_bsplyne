 53        self,
 54        spline: BSpline,
 55        ctrl_pts: np.ndarray[np.floating],
 56        E: float,
 57        nu: float,
 58        F_N: np.ndarray[np.floating] = np.zeros((3, 2, 3), dtype="float"),
 59    ):
 60        """
 61        Initialize the IGAPatch with the given parameters.
 62
 63        Parameters
 64        ----------
 65        spline : BSpline
 66            B-spline volume used as the patch.
 67        ctrl_pts : np.ndarray[np.floating]
 68            Control points defining the patch geometry.
 69        E : float
 70            Young's modulus of the material.
 71        nu : float
 72            Poisson's ratio of the material.
 73        F_N : np.ndarray[np.floating], optional
 74            Surfacic forces applied on the corresponding side of the patch.
 75            Its shape should be (3(param), 2(side), 3(phy)).
 76            By default np.zeros((3, 2, 3), dtype='float').
 77        """
 78        self.spline = spline
 79        self.ctrl_pts = ctrl_pts
 80        self.H = (
 81            np.array(
 82                [
 83                    [1 - nu, nu, nu, 0, 0, 0],
 84                    [nu, 1 - nu, nu, 0, 0, 0],
 85                    [nu, nu, 1 - nu, 0, 0, 0],
 86                    [0, 0, 0, (0.5 - nu), 0, 0],
 87                    [0, 0, 0, 0, (0.5 - nu), 0],
 88                    [0, 0, 0, 0, 0, (0.5 - nu)],
 89                ],
 90                dtype="float",
 91            )
 92            * E
 93            / ((1 + nu) * (1 - 2 * nu))

 99            xi_eta_zeta_pts_pds

101        self.F_N = (
102            F_N  # shape (3 (xi, eta, or zeta), 2 (front or back), 3 (vector (x, y, z)))

105    def jacobian(
106        self, dN_dXI: tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
107    ) -> tuple[
108        np.ndarray[np.floating], np.ndarray[np.floating], np.ndarray[np.floating]
109    ]:
110        """
111        Calculate the Jacobian matrix (derivative of the mesh wrt its isoparametric space),
112        its inverse, and its determinant.
113
114        Parameters
115        ----------
116        dN_dXI : tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
117            Tuple of sparse matrices representing the derivatives of shape
118            functions wrt the isoparametric space.
119            Contains dN_dxi, dN_deta and dN_dzeta.
120
121        Returns
122        -------
123        J : np.ndarray[np.floating]
124            Jacobian matrix, with shape (3(phy), 3(param), nb_intg_pts).
125        Jinv : np.ndarray[np.floating]
126            Inverse of the Jacobian matrix, with shape (3(param), 3(phy), nb_intg_pts).
127        detJ : np.ndarray[np.floating]
128            Determinant of the Jacobian matrix, with shape (nb_intg_pts,).
129        """
130        J = np.concatenate(
131            [(self.ctrl_pts.reshape((3, -1)) @ dN_dxi.T)[:, None] for dN_dxi in dN_dXI],
132            axis=1,
133        )  # shape(3(phy), 3(param), nb_intg_pts)
134        Jinv = np.linalg.inv(J.transpose((2, 0, 1))).transpose(
135            (1, 2, 0)
136        )  # shape(3(param), 3(phy), nb_intg_pts)
137        detJ = np.linalg.det(J.transpose((2, 0, 1)))  # vector of nb_intg_pts values
138        return J, Jinv, detJ

140    def grad_N(
141        self,
142        Jinv: np.ndarray[np.floating],
143        dN_dXI: tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix],
144    ) -> np.ndarray[np.object_]:
145        """
146        Calculate the gradient of shape functions with respect to physical space.
147
148        Parameters
149        ----------
150        Jinv : np.ndarray[np.floating]
151            Inverse of the Jacobian matrix, with shape (3(phy), 3(param), nb_intg_pts).
152        dN_dXI : tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
153            Tuple of sparse matrices representing the derivatives of shape
154            functions with respect to the isoparametric space.
155
156        Returns
157        -------
158        dN_dX : np.ndarray[np.object_]
159            Gradient of shape functions with respect to physical space.
160            Numpy array of shape (3(phy),) containing `sps.spmatrix` objects of shape (nb_intg_pts, nb_ctrl_pts).
161        """
162        dN_dX = np.array(
163            [
164                sum([dN_dXI[j].multiply(Jinv[j, i, :, None]) for j in range(3)])
165                for i in range(3)
166            ]
167        )
168        return dN_dX

170    def make_W(self, detJ: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
171        """
172        Compute the measure for intagrating in the physical space as ( abs(det(dX_dXI))*dxi*deta*dzeta ).
173        This is the element-wise product of the absolute value of detJ and the Kronecker product
174        of dxi, deta, and dzeta.
175
176        Parameters
177        ----------
178        detJ : np.ndarray[np.floating]
179            Array containing the determinant values.
180            Its shape is (nb_intg_pts,).
181
182        Returns
183        -------
184        W : np.ndarray[np.floating]
185            Array containing the measure for intagrating in the physical space.
186            Its shape is (nb_intg_pts,).
187        """
188        W = np.abs(detJ) * np.outer(np.outer(self.dxi, self.deta), self.dzeta).ravel()
189        return W

194
195        Returns
196        -------
197        K : sps.spmatrix
198            Stiffness matrix computed based on the given parameters and operations.
199        """
200        dN_dXI = tuple(self.spline.DN([self.xi, self.eta, self.zeta], k=1).tolist())
201        _, Jinv, detJ = self.jacobian(dN_dXI)
202        dN_dx, dN_dy, dN_dz = self.grad_N(Jinv, dN_dXI)
203        W = self.make_W(detJ)
204        # B = sps.bmat([[dN_dx, None, None],
205        #               [None, dN_dy, None],
206        #               [None, None, dN_dz],
207        #               [dN_dy, dN_dx, None],
208        #               [None, dN_dz, dN_dy],
209        #               [dN_dz, None, dN_dx]])
210
211        Z = sps.coo_matrix(dN_dx.shape, dtype="float")

213        Byy = sps.hstack((Z, dN_dy, Z))
214        Bzz = sps.hstack((Z, Z, dN_dz))
215        Bxy = sps.hstack((dN_dy, dN_dx, Z))
216        Byz = sps.hstack((Z, dN_dz, dN_dy))
217        Bxz = sps.hstack((dN_dz, Z, dN_dx))

220        K2 = Bzz.T @ sum((self.H[2, 0] * Bxx, self.H[2, 1] * Byy, self.H[2, 2] * Bzz, self.H[2, 3] * Bxy, self.H[2, 4] * Byz, self.H[2, 5] * Bxz)).T.multiply(W).T  # type: ignore
221        K3 = Bxy.T @ sum((self.H[3, 0] * Bxx, self.H[3, 1] * Byy, self.H[3, 2] * Bzz, self.H[3, 3] * Bxy, self.H[3, 4] * Byz, self.H[3, 5] * Bxz)).T.multiply(W).T  # type: ignore
222        K4 = Byz.T @ sum((self.H[4, 0] * Bxx, self.H[4, 1] * Byy, self.H[4, 2] * Bzz, self.H[4, 3] * Bxy, self.H[4, 4] * Byz, self.H[4, 5] * Bxz)).T.multiply(W).T  # type: ignore
223        K5 = Bxz.T @ sum((self.H[5, 0] * Bxx, self.H[5, 1] * Byy, self.H[5, 2] * Bzz, self.H[5, 3] * Bxy, self.H[5, 4] * Byz, self.H[5, 5] * Bxz)).T.multiply(W).T  # type: ignore
224        K: sps.spmatrix = sum((K0, K1, K2, K3, K4, K5))  # type: ignore
225        return K

227    def rhs(self) -> np.ndarray[np.floating]:
228        """
229        Calculate the right-hand side (rhs) vector for the IGAPatch.
230
231        Returns
232        -------
233        rhs : np.ndarray[np.floating]
234            The computed rhs vector based on the given parameters and operations.
235        """
236        rhs = np.zeros(self.ctrl_pts.size, dtype="float")
237        XI = [self.xi, self.eta, self.zeta]
238        dXI = [self.dxi, self.deta, self.dzeta]
239        for axis in range(3):
240            for side in range(2):
241                if not np.allclose(self.F_N[axis, side], 0):
242                    xi = [
243                        (
244                            XI[i]
245                            if i != axis
246                            else np.array(
247                                [
248                                    (
249                                        self.spline.bases[i].span[0]
250                                        if side == 0
251                                        else self.spline.bases[i].span[1]
252                                    )
253                                ]
254                            )
255                        )
256                        for i in range(3)
257                    ]
258                    N = self.spline.DN(xi)
259                    dN1, dN2 = [
260                        self.spline.DN(xi, k=k.tolist())
261                        for ax, k in enumerate(np.eye(3))
262                        if ax != axis
263                    ]
264                    dS = (
265                        np.linalg.norm(
266                            np.cross(
267                                self.ctrl_pts.reshape((3, -1)) @ dN1.T,
268                                self.ctrl_pts.reshape((3, -1)) @ dN2.T,
269                                axis=0,
270                            ),
271                            axis=0,
272                        )
273                        * np.outer(*[dXI[i] for i in range(3) if i != axis]).ravel()
274                    )
275                    intg = N.multiply(dS[:, None]).sum(axis=0).A.ravel()  # type: ignore
276                    rhs += np.outer(self.F_N[axis, side], intg).ravel()
277        return rhs

280        XI = [self.xi, self.eta, self.zeta]

305            np.cross(

309            ),
310            axis=0,
311        )
312        dXI_area = np.outer(*dxi).ravel()

316    def epsilon(
317        self, U: np.ndarray[np.floating], XI: list[np.ndarray[np.floating]]
318    ) -> np.ndarray[np.floating]:
319        """
320        Calculate the strain tensor (epsilon) for the IGAPatch based on
321        the displacement field `U` and the isoparametric coordinates `XI`.
322
323        Parameters
324        ----------
325        U : np.ndarray[np.floating]
326            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
327
328        XI : list[np.ndarray[np.floating]]
329            List of isoparametric coordinates for each direction xi, eta, and zeta.
330
331        Returns
332        -------
333        eps : np.ndarray[np.floating]
334            Strain tensor epsilon in voight notation
335            computed as a numpy array of shape (6, nb_param_pts).
336        """
337        for xi, basis in zip(XI, self.spline.bases):
338            xi[np.isclose(xi, basis.span[0], atol=1e-5)] += 1e-5
339            xi[np.isclose(xi, basis.span[1], atol=1e-5)] -= 1e-5
340        dN_dXI = tuple(self.spline.DN(XI, k=1).tolist())
341        _, Jinv, _ = self.jacobian(dN_dXI)  # dXI_dX
342        dN_dx, dN_dy, dN_dz = self.grad_N(Jinv, dN_dXI)
343        eps = np.array(
344            [
345                dN_dx @ U[0].ravel(),
346                dN_dy @ U[1].ravel(),
347                dN_dz @ U[2].ravel(),
348                0.5 * (dN_dy @ U[0].ravel() + dN_dx @ U[1].ravel()),
349                0.5 * (dN_dz @ U[1].ravel() + dN_dy @ U[2].ravel()),
350                0.5 * (dN_dz @ U[0].ravel() + dN_dx @ U[2].ravel()),
351            ]
352        )
353        return eps

355    def sigma(
356        self, U: np.ndarray[np.floating], XI: list[np.ndarray[np.floating]]
357    ) -> np.ndarray[np.floating]:
358        """
359        Calculate the stress tensor (sigma) for the IGAPatch based on
360        the displacement field `U` and the isoparametric coordinates `XI`.
361
362        Parameters
363        ----------
364        U : np.ndarray[np.floating]
365            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
366
367        XI : list[np.ndarray[np.floating]]
368            List of isoparametric coordinates for each direction xi, eta, and zeta.
369
370        Returns
371        -------
372        sig : np.ndarray[np.floating]
373            Stress tensor sigma in voight notation
374            computed as a numpy array of shape (6, nb_param_pts).
375        """
376        eps = self.epsilon(U, XI)
377        eps[3:] *= 2
378        sig = self.H @ eps
379        return sig

381    def sigma_eig(
382        self, U: np.ndarray[np.floating], XI: list[np.ndarray[np.floating]]
383    ) -> np.ndarray[np.floating]:
384        """
385        Calculate the eigenvalues of the stress tensor for the IGAPatch based on
386        the displacement field `U` and the isoparametric coordinates `XI`.
387
388        Parameters
389        ----------
390        U : np.ndarray[np.floating]
391            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
392
393        XI : list[np.ndarray[np.floating]]
394            List of isoparametric coordinates for each direction xi, eta, and zeta.
395
396        Returns
397        -------
398        sig_eig : np.ndarray[np.floating]
399            Eigenvalues of the stress tensor computed as a numpy array of shape (nb_param_pts, 3).
400        """
401        sig = self.sigma(U, XI)
402        sig = sig[np.array([[0, 3, 5], [3, 1, 4], [5, 4, 2]])]
403        sig_eig = np.linalg.eigvalsh(sig.transpose((2, 0, 1))).T
404        order = np.argsort(np.abs(sig_eig), axis=0)
405        sig_eig = sig_eig[order, np.arange(sig_eig.shape[1])[None]]
406        return sig_eig

408    def von_mises(
409        self, U: np.ndarray[np.floating], XI: list[np.ndarray[np.floating]]
410    ) -> np.ndarray[np.floating]:
411        """
412        Calculate the von Mises stress for the IGAPatch based on the displacement field `U`
413        and the isoparametric coordinates `XI`.
414
415        Parameters
416        ----------
417        U : np.ndarray[np.floating]
418            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
419
420        XI : list[np.ndarray[np.floating]]
421            List of isoparametric coordinates for each direction xi, eta, and zeta.
422
423        Returns
424        -------
425        vm : np.ndarray[np.floating]
426            Von Mises stress computed as a numpy array of shape (nb_param_pts,).
427        """
428        sig_eig = self.sigma_eig(U, XI)
429        vm = (1 / np.sqrt(2)) * np.linalg.norm(
430            sig_eig - np.roll(sig_eig, -1, axis=0), axis=0
431        )
432        return vm

106    def jacobian(
107        self, dN_dXI: tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
108    ) -> tuple[
109        np.ndarray[np.floating], np.ndarray[np.floating], np.ndarray[np.floating]
110    ]:
111        """
112        Calculate the Jacobian matrix (derivative of the mesh wrt its isoparametric space),
113        its inverse, and its determinant.
114
115        Parameters
116        ----------
117        dN_dXI : tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
118            Tuple of sparse matrices representing the derivatives of shape
119            functions wrt the isoparametric space.
120            Contains dN_dxi, dN_deta and dN_dzeta.
121
122        Returns
123        -------
124        J : np.ndarray[np.floating]
125            Jacobian matrix, with shape (3(phy), 3(param), nb_intg_pts).
126        Jinv : np.ndarray[np.floating]
127            Inverse of the Jacobian matrix, with shape (3(param), 3(phy), nb_intg_pts).
128        detJ : np.ndarray[np.floating]
129            Determinant of the Jacobian matrix, with shape (nb_intg_pts,).
130        """
131        J = np.concatenate(
132            [(self.ctrl_pts.reshape((3, -1)) @ dN_dxi.T)[:, None] for dN_dxi in dN_dXI],
133            axis=1,
134        )  # shape(3(phy), 3(param), nb_intg_pts)
135        Jinv = np.linalg.inv(J.transpose((2, 0, 1))).transpose(
136            (1, 2, 0)
137        )  # shape(3(param), 3(phy), nb_intg_pts)
138        detJ = np.linalg.det(J.transpose((2, 0, 1)))  # vector of nb_intg_pts values
139        return J, Jinv, detJ

228    def rhs(self) -> np.ndarray[np.floating]:
229        """
230        Calculate the right-hand side (rhs) vector for the IGAPatch.
231
232        Returns
233        -------
234        rhs : np.ndarray[np.floating]
235            The computed rhs vector based on the given parameters and operations.
236        """
237        rhs = np.zeros(self.ctrl_pts.size, dtype="float")
238        XI = [self.xi, self.eta, self.zeta]
239        dXI = [self.dxi, self.deta, self.dzeta]
240        for axis in range(3):
241            for side in range(2):
242                if not np.allclose(self.F_N[axis, side], 0):
243                    xi = [
244                        (
245                            XI[i]
246                            if i != axis
247                            else np.array(
248                                [
249                                    (
250                                        self.spline.bases[i].span[0]
251                                        if side == 0
252                                        else self.spline.bases[i].span[1]
253                                    )
254                                ]
255                            )
256                        )
257                        for i in range(3)

260                    dN1, dN2 = [
261                        self.spline.DN(xi, k=k.tolist())

319    ) -> np.ndarray[np.floating]:
320        """
321        Calculate the strain tensor (epsilon) for the IGAPatch based on
322        the displacement field `U` and the isoparametric coordinates `XI`.
323
324        Parameters
325        ----------
326        U : np.ndarray[np.floating]
327            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
328
329        XI : list[np.ndarray[np.floating]]
330            List of isoparametric coordinates for each direction xi, eta, and zeta.
331
332        Returns
333        -------
334        eps : np.ndarray[np.floating]
335            Strain tensor epsilon in voight notation
336            computed as a numpy array of shape (6, nb_param_pts).
337        """
338        for xi, basis in zip(XI, self.spline.bases):
339            xi[np.isclose(xi, basis.span[0], atol=1e-5)] += 1e-5

341        dN_dXI = tuple(self.spline.DN(XI, k=1).tolist())

343        dN_dx, dN_dy, dN_dz = self.grad_N(Jinv, dN_dXI)
344        eps = np.array(
345            [
346                dN_dx @ U[0].ravel(),
347                dN_dy @ U[1].ravel(),
348                dN_dz @ U[2].ravel(),
349                0.5 * (dN_dy @ U[0].ravel() + dN_dx @ U[1].ravel()),
350                0.5 * (dN_dz @ U[1].ravel() + dN_dy @ U[2].ravel()),
351                0.5 * (dN_dz @ U[0].ravel() + dN_dx @ U[2].ravel()),
352            ]
353        )
