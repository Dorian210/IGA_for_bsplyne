 81    def jacobian(self, dN_dXI: tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]) -> tuple[npt.NDArray[np.float_], npt.NDArray[np.float_], npt.NDArray[np.float_]]:
 82        """
 83        Calculate the Jacobian matrix (derivative of the mesh wrt its isoparametric space), 
 84        its inverse, and its determinant.
 85
 86        Parameters
 87        ----------
 88        dN_dXI : tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
 89            Tuple of sparse matrices representing the derivatives of shape 
 90            functions wrt the isoparametric space.
 91            Contains dN_dxi, dN_deta and dN_dzeta.
 92
 93        Returns
 94        -------
 95        J : npt.NDArray[np.float_]
 96            Jacobian matrix, with shape (3(phy), 3(param), nb_intg_pts).
 97        Jinv : npt.NDArray[np.float_]
 98            Inverse of the Jacobian matrix, with shape (3(param), 3(phy), nb_intg_pts).
 99        detJ : npt.NDArray[np.float_]
100            Determinant of the Jacobian matrix, with shape (nb_intg_pts,).
101        """
102        J = np.concatenate([(self.ctrl_pts.reshape((3, -1))@dN_dxi.T)[:, None] for dN_dxi in dN_dXI], axis=1) # shape(3(phy), 3(param), nb_intg_pts)
103        Jinv = np.linalg.inv(J.transpose((2, 0, 1))).transpose((1, 2, 0)) # shape(3(param), 3(phy), nb_intg_pts)
104        detJ = np.linalg.det(J.transpose((2, 0, 1))) # vector of nb_intg_pts values
105        return J, Jinv, detJ

107    def grad_N(self, Jinv: npt.NDArray[np.float_], dN_dXI: tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]) -> npt.NDArray[np.object_]:
108        """
109        Calculate the gradient of shape functions with respect to physical space.
110
111        Parameters
112        ----------
113        Jinv : npt.NDArray[np.float_]
114            Inverse of the Jacobian matrix, with shape (3(phy), 3(param), nb_intg_pts).
115        dN_dXI : tuple[sps.spmatrix, sps.spmatrix, sps.spmatrix]
116            Tuple of sparse matrices representing the derivatives of shape 
117            functions with respect to the isoparametric space.
118
119        Returns
120        -------
121        dN_dX : npt.NDArray[np.object_]
122            Gradient of shape functions with respect to physical space.
123            Numpy array of shape (3(phy),) containing `sps.spmatrix` objects of shape (nb_intg_pts, nb_ctrl_pts).
124        """
125        dN_dX = np.array([sum([dN_dXI[j].multiply(Jinv[j, i, :, None]) for j in range(3)]) for i in range(3)])
126        return dN_dX

128    def make_W(self, detJ: npt.NDArray[np.float_]) -> npt.NDArray[np.float_]:
129        """
130        Compute the measure for intagrating in the physical space as ( abs(det(dX_dXI))*dxi*deta*dzeta ).
131        This is the element-wise product of the absolute value of detJ and the Kronecker product 
132        of dxi, deta, and dzeta.
133
134        Parameters
135        ----------
136        detJ : npt.NDArray[np.float_]
137            Array containing the determinant values.
138            Its shape is (nb_intg_pts,).
139
140        Returns
141        -------
142        W : npt.NDArray[np.float_]
143            Array containing the measure for intagrating in the physical space.
144            Its shape is (nb_intg_pts,).
145        """
146        W = np.abs(detJ)*np.outer(np.outer(self.dxi, self.deta), self.dzeta).ravel()
147        return W

149    def stiffness(self) -> sps.spmatrix:
150        """
151        Calculate the stiffness matrix for the IGAPatch.
152
153        Returns
154        -------
155        K : sps.spmatrix
156            Stiffness matrix computed based on the given parameters and operations.
157        """
158        dN_dXI = tuple(self.spline.DN([self.xi, self.eta, self.zeta], k=1).tolist())
159        _, Jinv, detJ = self.jacobian(dN_dXI)
160        dN_dx, dN_dy, dN_dz = self.grad_N(Jinv, dN_dXI)
161        W = self.make_W(detJ)
162        # B = sps.bmat([[dN_dx, None, None], 
163        #               [None, dN_dy, None], 
164        #               [None, None, dN_dz], 
165        #               [dN_dy, dN_dx, None], 
166        #               [None, dN_dz, dN_dy], 
167        #               [dN_dz, None, dN_dx]])
168        
169        Z = sps.coo_matrix(dN_dx.shape, dtype='float')
170        Bxx = sps.hstack((dN_dx, Z, Z))
171        Byy = sps.hstack((Z, dN_dy, Z))
172        Bzz = sps.hstack((Z, Z, dN_dz))
173        Bxy = sps.hstack((dN_dy, dN_dx, Z))
174        Byz = sps.hstack((Z, dN_dz, dN_dy))
175        Bxz = sps.hstack((dN_dz, Z, dN_dx))
176        K0 = Bxx.T @ sum((self.H[0, 0]*Bxx, self.H[0, 1]*Byy, self.H[0, 2]*Bzz, self.H[0, 3]*Bxy, self.H[0, 4]*Byz, self.H[0, 5]*Bxz)).T.multiply(W).T # type: ignore
177        K1 = Byy.T @ sum((self.H[1, 0]*Bxx, self.H[1, 1]*Byy, self.H[1, 2]*Bzz, self.H[1, 3]*Bxy, self.H[1, 4]*Byz, self.H[1, 5]*Bxz)).T.multiply(W).T # type: ignore
178        K2 = Bzz.T @ sum((self.H[2, 0]*Bxx, self.H[2, 1]*Byy, self.H[2, 2]*Bzz, self.H[2, 3]*Bxy, self.H[2, 4]*Byz, self.H[2, 5]*Bxz)).T.multiply(W).T # type: ignore
179        K3 = Bxy.T @ sum((self.H[3, 0]*Bxx, self.H[3, 1]*Byy, self.H[3, 2]*Bzz, self.H[3, 3]*Bxy, self.H[3, 4]*Byz, self.H[3, 5]*Bxz)).T.multiply(W).T # type: ignore
180        K4 = Byz.T @ sum((self.H[4, 0]*Bxx, self.H[4, 1]*Byy, self.H[4, 2]*Bzz, self.H[4, 3]*Bxy, self.H[4, 4]*Byz, self.H[4, 5]*Bxz)).T.multiply(W).T # type: ignore
181        K5 = Bxz.T @ sum((self.H[5, 0]*Bxx, self.H[5, 1]*Byy, self.H[5, 2]*Bzz, self.H[5, 3]*Bxy, self.H[5, 4]*Byz, self.H[5, 5]*Bxz)).T.multiply(W).T # type: ignore
182        K: sps.spmatrix = sum((K0, K1, K2, K3, K4, K5)) # type: ignore
183        return K

185    def rhs(self) -> npt.NDArray[np.float_]:
186        """
187        Calculate the right-hand side (rhs) vector for the IGAPatch.
188
189        Returns
190        -------
191        rhs : npt.NDArray[np.float_]
192            The computed rhs vector based on the given parameters and operations.
193        """
194        rhs = np.zeros(self.ctrl_pts.size, dtype='float')
195        XI = [self.xi, self.eta, self.zeta]
196        dXI = [self.dxi, self.deta, self.dzeta]
197        for axis in range(3):
198            for side in range(2):
199                if not np.allclose(self.F_N[axis, side], 0):
200                    xi = [XI[i] if i!=axis else np.array([self.spline.bases[i].span[0] if side==0 else self.spline.bases[i].span[1]]) for i in range(3)]
201                    N = self.spline.DN(xi)
202                    dN1, dN2 = [self.spline.DN(xi, k=k.tolist()) for ax, k in enumerate(np.eye(3)) if ax!=axis]
203                    dS = np.linalg.norm(np.cross(self.ctrl_pts.reshape((3, -1)) @ dN1.T, self.ctrl_pts.reshape((3, -1)) @ dN2.T, axis=0), axis=0)*np.outer(*[dXI[i] for i in range(3) if i!=axis]).ravel()
204                    intg = N.multiply(dS[:, None]).sum(axis=0).A.ravel() # type: ignore
205                    rhs += np.outer(self.F_N[axis, side], intg).ravel()
206        return rhs

208    def epsilon(self, U: npt.NDArray[np.float_], XI: list[npt.NDArray[np.float_]]) -> npt.NDArray[np.float_]:
209        """
210        Calculate the strain tensor (epsilon) for the IGAPatch based on 
211        the displacement field `U` and the isoparametric coordinates `XI`.
212
213        Parameters
214        ----------
215        U : npt.NDArray[np.float_]
216            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
217
218        XI : list[npt.NDArray[np.float_]]
219            List of isoparametric coordinates for each direction xi, eta, and zeta.
220
221        Returns
222        -------
223        eps : npt.NDArray[np.float_]
224            Strain tensor epsilon in voight notation 
225            computed as a numpy array of shape (6, nb_param_pts).
226        """
227        for xi, basis in zip(XI, self.spline.bases):
228            xi[np.isclose(xi, basis.span[0], atol=1e-5)] += 1e-5
229            xi[np.isclose(xi, basis.span[1], atol=1e-5)] -= 1e-5
230        dN_dXI = tuple(self.spline.DN(XI, k=1).tolist())
231        _, Jinv, _ = self.jacobian(dN_dXI) # dXI_dX
232        dN_dx, dN_dy, dN_dz = self.grad_N(Jinv, dN_dXI)
233        eps = np.array([dN_dx @ U[0].ravel(), 
234                        dN_dy @ U[1].ravel(), 
235                        dN_dz @ U[2].ravel(), 
236                        0.5*(dN_dy @ U[0].ravel() + dN_dx @ U[1].ravel()), 
237                        0.5*(dN_dz @ U[1].ravel() + dN_dy @ U[2].ravel()), 
238                        0.5*(dN_dz @ U[0].ravel() + dN_dx @ U[2].ravel())])
239        return eps

241    def sigma(self, U: npt.NDArray[np.float_], XI: list[npt.NDArray[np.float_]]) -> npt.NDArray[np.float_]:
242        """
243        Calculate the stress tensor (sigma) for the IGAPatch based on 
244        the displacement field `U` and the isoparametric coordinates `XI`.
245
246        Parameters
247        ----------
248        U : npt.NDArray[np.float_]
249            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
250
251        XI : list[npt.NDArray[np.float_]]
252            List of isoparametric coordinates for each direction xi, eta, and zeta.
253
254        Returns
255        -------
256        sig : npt.NDArray[np.float_]
257            Stress tensor sigma in voight notation 
258            computed as a numpy array of shape (6, nb_param_pts).
259        """
260        eps = self.epsilon(U, XI)
261        eps[3:] *= 2
262        sig = self.H @ eps
263        return sig

265    def sigma_eig(self, U: npt.NDArray[np.float_], XI: list[npt.NDArray[np.float_]]) -> npt.NDArray[np.float_]:
266        """
267        Calculate the eigenvalues of the stress tensor for the IGAPatch based on 
268        the displacement field `U` and the isoparametric coordinates `XI`.
269
270        Parameters
271        ----------
272        U : npt.NDArray[np.float_]
273            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
274
275        XI : list[npt.NDArray[np.float_]]
276            List of isoparametric coordinates for each direction xi, eta, and zeta.
277
278        Returns
279        -------
280        sig_eig : npt.NDArray[np.float_]
281            Eigenvalues of the stress tensor computed as a numpy array of shape (nb_param_pts, 3).
282        """
283        sig = self.sigma(U, XI)
284        sig = sig[np.array([[0, 3, 5], 
285                            [3, 1, 4], 
286                            [5, 4, 2]])]
287        sig_eig = np.linalg.eigvalsh(sig.transpose((2, 0, 1))).T
288        order = np.argsort(np.abs(sig_eig), axis=0)
289        sig_eig = sig_eig[order, np.arange(sig_eig.shape[1])[None]]
290        return sig_eig

292    def von_mises(self, U: npt.NDArray[np.float_], XI: list[npt.NDArray[np.float_]]) -> npt.NDArray[np.float_]:
293        """
294        Calculate the von Mises stress for the IGAPatch based on the displacement field `U` 
295        and the isoparametric coordinates `XI`.
296
297        Parameters
298        ----------
299        U : npt.NDArray[np.float_]
300            Displacement field as a numpy array of shape (3(phy), nb_ctrl_pts).
301
302        XI : list[npt.NDArray[np.float_]]
303            List of isoparametric coordinates for each direction xi, eta, and zeta.
304
305        Returns
306        -------
307        vm : npt.NDArray[np.float_]
308            Von Mises stress computed as a numpy array of shape (nb_param_pts,).
309        """
310        sig_eig = self.sigma_eig(U, XI)
311        vm = (1/np.sqrt(2))*np.linalg.norm(sig_eig - np.roll(sig_eig, -1, axis=0), axis=0)
312        return vm

187        """
188        Calculate the right-hand side (rhs) vector for the IGAPatch.
189
190        Returns
191        -------
192        rhs : npt.NDArray[np.float_]
193            The computed rhs vector based on the given parameters and operations.
194        """
195        rhs = np.zeros(self.ctrl_pts.size, dtype='float')
196        XI = [self.xi, self.eta, self.zeta]
197        dXI = [self.dxi, self.deta, self.dzeta]
198        for axis in range(3):
199            for side in range(2):
200                if not np.allclose(self.F_N[axis, side], 0):
201                    xi = [XI[i] if i!=axis else np.array([self.spline.bases[i].span[0] if side==0 else self.spline.bases[i].span[1]]) for i in range(3)]
202                    N = self.spline.DN(xi)
203                    dN1, dN2 = [self.spline.DN(xi, k=k.tolist()) for ax, k in enumerate(np.eye(3)) if ax!=axis]
204                    dS = np.linalg.norm(np.cross(self.ctrl_pts.reshape((3, -1)) @ dN1.T, self.ctrl_pts.reshape((3, -1)) @ dN2.T, axis=0), axis=0)*np.outer(*[dXI[i] for i in range(3) if i!=axis]).ravel()

206                    rhs += np.outer(self.F_N[axis, side], intg).ravel()

210        """
211        Calculate the strain tensor (epsilon) for the IGAPatch based on 
212        the displacement field `U` and the isoparametric coordinates `XI`.
213
214        Parameters
215        ----------
216        U : npt.NDArray[np.float_]

228        for xi, basis in zip(XI, self.spline.bases):
229            xi[np.isclose(xi, basis.span[0], atol=1e-5)] += 1e-5

234        eps = np.array([dN_dx @ U[0].ravel(), 
235                        dN_dy @ U[1].ravel(), 
236                        dN_dz @ U[2].ravel(), 
237                        0.5*(dN_dy @ U[0].ravel() + dN_dx @ U[1].ravel()), 
238                        0.5*(dN_dz @ U[1].ravel() + dN_dy @ U[2].ravel()), 
239                        0.5*(dN_dz @ U[0].ravel() + dN_dx @ U[2].ravel())])
240        return eps
