  9class Dirichlet:
 10    """
 11    Affine representation of linear Dirichlet constraints.
 12
 13    This class represents Dirichlet boundary conditions through an affine
 14    transformation between a reduced vector of free parameters (`dof`)
 15    and the full physical displacement vector (`u`):
 16
 17        u = C @ dof + k
 18
 19    where:
 20
 21    - `u` is the full displacement vector satisfying all imposed constraints.
 22    - `dof` is the reduced vector of unconstrained degrees of freedom.
 23    - `C` is a sparse matrix whose columns form a basis of admissible variations.
 24    - `k` is a particular displacement vector satisfying the constraints.
 25
 26    This representation allows:
 27    - elimination of constrained DOFs,
 28    - support for general linear multi-point constraints.
 29
 30    Attributes
 31    ----------
 32    C : sps.csc_matrix of shape (n_full_dofs, n_free_dofs)
 33        Sparse matrix defining the linear mapping from reduced DOFs to full DOFs.
 34    k : np.ndarray of shape (n_full_dofs,)
 35        Particular solution ensuring that `u` satisfies the Dirichlet constraints.
 36    """
 37
 38    C: sps.csc_matrix
 39    k: np.ndarray[np.floating]

 41    def __init__(self, C: sps.spmatrix, k: np.ndarray[np.floating]):
 42        """
 43        Initialize an affine Dirichlet constraint representation.
 44
 45        Parameters
 46        ----------
 47        C : sps.spmatrix
 48            Sparse matrix of shape (n_full_dofs, n_free_dofs) defining
 49            the linear mapping from reduced DOFs to full DOFs.
 50        k : np.ndarray[np.floating]
 51            Vector of shape (n_full_dofs,) defining the affine offset.

 53        Notes
 54        -----
 55        The matrix `C` is internally converted to CSC format for efficient
 56        matrix-vector products.
 57        """
 58        self.C = C.tocsc()
 59        self.k = k

 90    def lock_disp_inds(cls, inds: np.ndarray[np.integer], k: np.ndarray[np.floating]):
 91        """
 92        Create Dirichlet constraints by prescribing displacement values at selected DOFs.
 93
 94        This method enforces pointwise Dirichlet conditions of the form:
 95
 96            u[i] = k[i]    for i in inds
 97
 98        All other DOFs remain unconstrained.
 99
100        Parameters
101        ----------
102        inds : np.ndarray[np.integer]
103            Indices of the displacement vector `u` to be constrained.
104        k : np.ndarray[np.floating]
105            Target displacement vector of shape (n_full_dofs,). Only the
106            values at `inds` are enforced.
107
108        Returns
109        -------
110        dirichlet : Dirichlet
111            `Dirichlet` instance representing the prescribed displacement constraints.
112
113        Notes
114        -----
115        Constrained DOFs are removed from the reduced DOF vector. The resulting
116        number of reduced DOFs will therefore be smaller than `k.size`.
117        """
118        C = sps.eye(k.size)
119        dirichlet = cls(C, k.copy())
120        dirichlet.set_u_inds_vals(inds, dirichlet.k[inds])
121        return dirichlet

123    def set_u_inds_vals(
124        self, inds: np.ndarray[np.integer], vals: np.ndarray[np.floating]
125    ):
126        """
127        Impose pointwise Dirichlet conditions by prescribing displacement values
128        at selected full DOFs.
129
130        This method enforces constraints of the form:
131
132            u[i] = vals[j]     for i = inds[j]
133
134        by modifying the affine mapping:
135
136            u = C @ dof + k
137
138        The modification removes the corresponding admissible variations in `C`
139        and updates `k` so that the prescribed values are always satisfied.
140        As a result, constrained DOFs are eliminated from the reduced DOF vector.
141
142        Parameters
143        ----------
144        inds : np.ndarray[np.integer]
145            Indices of the full displacement vector `u` to constrain.
146        vals : np.ndarray[np.floating]
147            Prescribed displacement values associated with `inds`.
148            Must have the same length as `inds`.
149
150        Notes
151        -----
152        - Rows of `C` corresponding to constrained DOFs are zeroed.
153        - Columns of `C` that become unused are removed, which may reduce
154          the number of reduced DOFs.
155        - The affine offset `k` is updated to enforce the prescribed values.
156        - This operation modifies the current Dirichlet object in place.
157        """
158
159        def zero_rows(C: sps.coo_matrix, rows: np.ndarray[np.integer]):
160            rm = np.hstack([(C.row == row).nonzero()[0] for row in rows])
161            row = np.delete(C.row, rm)
162            col = np.delete(C.col, rm)
163            data = np.delete(C.data, rm)
164            C = sps.coo_matrix((data, (row, col)), shape=(C.shape))
165            return C
166
167        def remove_zero_cols(C: sps.coo_matrix):
168            unique, inverse = np.unique(C.col, return_inverse=True)
169            C.col = np.arange(unique.size)[inverse]
170            C._shape = (C.shape[0], unique.size)
171
172        if inds.size != 0:
173            C_coo = zero_rows(self.C.tocoo(), inds)
174            remove_zero_cols(C_coo)
175            self.C = C_coo.tocsc()
176            self.k[inds] = vals

178    def slave_reference_linear_relation(
179        self,
180        slaves: np.ndarray[int],
181        references: np.ndarray[int],
182        coefs: Union[np.ndarray[float], None] = None,
183    ):
184        """
185        Impose linear multi-point constraints between full DOFs.
186
187        This method enforces relations of the form:
188
189            u[slave_i] = sum_j coefs[i, j] * u[references[i, j]]
190
191        by modifying the affine mapping:
192
193            u = C @ dof + k
194
195        Slave DOFs are expressed as linear combinations of reference DOFs.
196        The corresponding admissible variations are propagated into `C`
197        and `k`, effectively eliminating slave DOFs from the reduced parameter
198        vector while preserving the imposed constraints.
199
200        Parameters
201        ----------
202        slaves : np.ndarray[int] of shape (n_slaves,)
203            Indices of DOFs that are constrained (slave DOFs).
204        references : np.ndarray[int] of shape (n_slaves, n_refs)
205            Reference DOF indices controlling each slave DOF.
206            Row `i` contains the references associated with `slaves[i]`.
207        coefs : np.ndarray[float], optional, shape (n_slaves, n_refs)
208            Linear combination coefficients linking references to slaves.
209            If None, slaves are defined as the average of their references.
210
211        Notes
212        -----
213        - Slave DOFs are removed from the reduced DOF vector.
214        - The constraint propagation accounts for hierarchical dependencies
215          between slave DOFs using a topological ordering.
216        - This operation is typically faster than using `DirichletConstraintHandler`,
217          as it directly modifies the affine mapping without constructing
218          intermediate constraint objects.
219        - Cyclic dependencies between slaves are not supported and will raise an error.
220        - This operation modifies the current Dirichlet object in place.
221
222        Examples
223        --------
224        To impose u[i] = 0.5 * (u[j] + u[k]):
225
226        >>> slaves = np.array([i])
227        >>> references = np.array([[j, k]])
228        >>> coefs = np.array([[0.5, 0.5]])
229        >>> dirichlet.slave_reference_linear_relation(slaves, references, coefs)
230        """
231
232        if coefs is None:
233            coefs = np.ones(references.shape, dtype="float") / references.shape[1]
234
235        sorted_slaves = slave_reference_linear_relation_sort(slaves, references)
236
237        C = self.C.tocsr()
238        rows, cols, data, self.k = slave_reference_linear_relation_inner(
239            C.indices,
240            C.indptr,
241            C.data,
242            self.k,
243            slaves,
244            references,
245            coefs,
246            sorted_slaves,
247        )
248        C = sps.coo_matrix((data, (rows, cols)), shape=C.shape)
249
250        unique, inverse = np.unique(C.col, return_inverse=True)
251        C.col = np.arange(unique.size)[inverse]
252        C._shape = (C.shape[0], unique.size)
253        self.C = C.tocsc()

255    def u_du_ddof(
256        self, dof: np.ndarray[np.floating]
257    ) -> tuple[np.ndarray[np.floating], sps.csc_matrix]:
258        """
259        Evaluate the displacement field and its derivative with respect to the reduced DOFs.
260
261        The displacement field is obtained from the affine mapping:
262
263            u = C @ dof + k
264
265        Since the mapping is affine, the derivative of `u` with respect to `dof`
266        is constant and equal to `C`.
267
268        Parameters
269        ----------
270        dof : np.ndarray[np.floating] of shape (n_dof,)
271            Reduced degrees of freedom.
272
273        Returns
274        -------
275        u : np.ndarray[np.floating] of shape (n_u,)
276            Displacement field.
277        du_ddof : sps.csc_matrix of shape (n_u, n_dof)
278            Jacobian of `u` with respect to `dof` : the matrix `C`.
279
280        Notes
281        -----
282        The returned derivative is independent of `dof` because the mapping is affine.
283        """
284        u = self.C @ dof + self.k
285        du_ddof = self.C
286        return u, du_ddof

288    def u(self, dof: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
289        """
290        Evaluate the displacement field from reduced DOFs.
291
292        The displacement field is computed using the affine mapping:
293
294            u = C @ dof + k
295
296        Parameters
297        ----------
298        dof : np.ndarray[np.floating]
299            Reduced degrees of freedom. Can be either:
300
301            - shape (n_dof,)
302            - shape (..., n_dof) for batched evaluation
303
304        Returns
305        -------
306        u : np.ndarray[np.floating]
307            Displacement field with shape:
308
309            - (n_u,) if `dof` is 1D
310            - (..., n_u) if `dof` is batched
311
312        Notes
313        -----
314        This method supports vectorized evaluation over multiple DOF vectors.
315        """
316        if dof.ndim == 1:
317            u = self.C @ dof + self.k
318        else:
319            m, n = self.C.shape
320            dof_shape = dof.shape[:-1]
321            u = (self.C @ dof.reshape((-1, n)).T + self.k.reshape((m, 1))).T.reshape(
322                (*dof_shape, m)
323            )
324        return u

326    def dof_lsq(self, u: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
327        """
328        Compute reduced DOFs from a displacement field using a least-squares projection.
329
330        This method computes `dof` such that:
331
332            u  C @ dof + k
333
334        by solving the normal equations:
335
336            (C C) dof = C (u - k)
337
338        Parameters
339        ----------
340        u : np.ndarray[np.floating]
341            Displacement field. Can be either:
342
343            - shape (n_u,)
344            - shape (..., n_u) for batched projection
345
346        Returns
347        -------
348        dof : np.ndarray[np.floating]
349            Reduced degrees of freedom with shape:
350
351            - (n_dof,) if `u` is 1D
352            - (..., n_dof) if `u` is batched
353
354        Notes
355        -----
356        This operation performs a least-squares inversion of the affine mapping.
357        If `C` does not have full column rank, the solution corresponds to a
358        minimum-norm least-squares solution.
359
360        The system `(C C)` is solved using a sparse linear solver.
361        """
362        if u.ndim == 1:
363            dof = solve_sparse(self.C.T @ self.C, self.C.T @ (u - self.k))
364        else:
365            m, n = self.C.shape
366            u_shape = u.shape[:-1]
367            dof = solve_sparse(
368                self.C.T @ self.C, (u.reshape((-1, m)) - self.k[None]) @ self.C
369            ).reshape((*u_shape, n))
370        return dof

373@nb.njit(cache=True)
374def slave_reference_linear_relation_sort(
375    slaves: np.ndarray[int], references: np.ndarray[int]
376) -> np.ndarray[int]:
377    """
378    Sorts the slave nodes based on reference indices to respect
379    hierarchical dependencies (each slave is processed after its references).
380
381    Parameters
382    ----------
383    slaves : np.ndarray[int]
384        Array of slave indices.
385    references : np.ndarray[int]
386        2D array where each row contains the reference indices controlling a slave.
387
388    Returns
389    -------
390    sorted_slaves : np.ndarray[int]
391        Array of slave indices sorted based on dependencies.
392    """
393    slaves_set = set(slaves)
394    graph = {s: nb.typed.List.empty_list(nb.int64) for s in slaves}
395    indegree = {s: 0 for s in slaves}
396
397    # Building the graph
398    for i in range(len(slaves)):
399        for reference in references[i]:
400            if reference in slaves_set:
401                indegree[reference] += 1
402                graph[slaves[i]].append(reference)
403
404    # Queue for slaves with no dependencies (in-degree 0)
405    queue = [s for s in slaves if indegree[s] == 0]
406
407    # Topological sorting via BFS
408    sorted_slaves = np.empty(len(slaves), dtype="int")
409    i = 0
410    while queue:
411        slave = queue.pop(0)
412        sorted_slaves[i] = slave
413        i += 1
414        for dependent_slave in graph[slave]:
415            indegree[dependent_slave] -= 1
416            if indegree[dependent_slave] == 0:
417                queue.append(dependent_slave)
418
419    if i != len(slaves):
420        raise ValueError("Cyclic dependency detected in slaves.")
421
422    sorted_slaves = sorted_slaves[::-1]
423    return sorted_slaves

426@nb.njit(cache=True)
427def slave_reference_linear_relation_inner(
428    indices: np.ndarray[int],
429    indptr: np.ndarray[int],
430    data: np.ndarray[float],
431    k: np.ndarray[float],
432    slaves: np.ndarray[int],
433    references: np.ndarray[int],
434    coefs: np.ndarray[float],
435    sorted_slaves: np.ndarray[int],
436) -> tuple[np.ndarray[int], np.ndarray[int], np.ndarray[float], np.ndarray[float]]:
437    """
438    Applies slave-reference relations directly to CSR matrix arrays.
439
440    Parameters
441    ----------
442    indices : np.ndarray[int]
443        Column indices of CSR matrix.
444    indptr : np.ndarray[int]
445        Row pointers of CSR matrix.
446    data : np.ndarray[float]
447        Nonzero values of CSR matrix.
448    k : np.ndarray[float]
449        Vector to be updated.
450    slaves : np.ndarray[int]
451        Array of slave indices.
452    references : np.ndarray[int]
453        2D array where each row contains the reference indices controlling a slave.
454    coefs : np.ndarray[float]
455        2D array of coefficients defining the linear relationship between references and
456        slaves.
457    sorted_slaves : np.ndarray[int]
458        Array of slave indices sorted in topological order.
459
460    Returns
461    -------
462    rows : np.ndarray[int]
463        Updated row indices of COO matrix.
464    cols : np.ndarray[int]
465        Updated column indices of COO matrix.
466    data : np.ndarray[float]
467        Updated nonzero values of COO matrix.
468    k : np.ndarray[float]
469        Updated vector.
470    """
471
472    # Convert CSR to list of dicts
473    dict_list = [
474        {indices[j]: data[j] for j in range(indptr[i], indptr[i + 1])}
475        for i in range(indptr.size - 1)
476    ]
477
478    # Apply linear relation
479    slaves_to_index = {s: i for i, s in enumerate(slaves)}
480    for slave in sorted_slaves:
481        i = slaves_to_index[slave]
482        new_row = {}
483        for reference_ind, coef in zip(references[i], coefs[i]):
484            reference_row = dict_list[reference_ind]
485            for ind, val in reference_row.items():
486                if ind in new_row:
487                    new_row[ind] += coef * val
488                else:
489                    new_row[ind] = coef * val
490        new_row = {ind: val for ind, val in new_row.items() if val != 0}
491        dict_list[slave] = new_row
492        k[slave] = np.dot(k[references[i]], coefs[i])
493
494    # Convert list of dicts to COO
495    nnz = sum([len(row) for row in dict_list])
496    rows = np.empty(nnz, dtype=np.int32)
497    cols = np.empty(nnz, dtype=np.int32)
498    data = np.empty(nnz, dtype=np.float64)
499    pos = 0
500    for i, row in enumerate(dict_list):
501        for j, val in row.items():
502            rows[pos] = i
503            cols[pos] = j
504            data[pos] = val
505            pos += 1
506
507    return rows, cols, data, k

511class DirichletConstraintHandler:
512    """
513    Accumulate affine linear constraints and construct the associated Dirichlet mapping.
514
515    This class is designed to impose linear affine constraints of the form:
516
517        D @ u = c
518
519    where `u` is the full displacement (or state) vector. Constraints are progressively
520    accumulated and, once fully specified, converted into an affine parametrization of
521    the admissible solution space:
522
523        u = C @ dof + k
524
525    where:
526
527    - `C` is a basis of the nullspace of `D` (i.e. `D @ C = 0`) obtained by
528    QR decomposition,
529    - `k` is a particular solution satisfying the constraints,
530    obtained through the QR decomposition which helps solve:
531
532        D k = c
533
534    - `dof` is a reduced vector of unconstrained degrees of freedom.
535
536    The main purpose of this class is to provide a flexible and robust interface
537    to define constraints before constructing a `Dirichlet` object representing
538    the reduced parametrization.
539
540    Typical workflow
541    ----------------
542    1. Create a handler with the number of physical DOFs (`u`).
543    2. Add constraint equations using helper methods.
544    3. **Ensure that all reference DOFs (e.g., translations or rotations introduced for
545    rigid-body relations) are fully constrained before computing `C` and `k`.**
546    4. Build the reduced representation `(C, k)` or directly create a `Dirichlet` object.
547
548    This separation allows constraints to be assembled incrementally and validated
549    before generating the final affine mapping.
550
551    Attributes
552    ----------
553    nb_dofs_init : int
554        Number of DOFs in the original unconstrained system (physical DOFs `u`).
555    lhs : sps.spmatrix
556        Accumulated constraint matrix `D`.
557    rhs : np.ndarray[np.floating]
558        Accumulated right-hand side vector `c`.
559
560    Notes
561    -----
562    - The constraint system may include additional reference DOFs introduced
563    to express kinematic relations or rigid-body behaviors.
564    - **All reference DOFs must be fully constrained before computing `C` and `k`;**
565    otherwise, DOFs that lie in the kernel of `D` cannot be controlled and imposed values
566    (e.g., prescribed translations) may not appear in the resulting solution `k`.
567    - The resulting affine mapping guarantees that any generated vector `u`
568    satisfies all imposed constraints.
569    """
570
571    nb_dofs_init: int
572    lhs: sps.spmatrix
573    rhs: np.ndarray[np.floating]
574
575    def __init__(self, nb_dofs_init: int):
576        """
577        Initialize a Dirichlet constraint handler.
578
579        Parameters
580        ----------
581        nb_dofs_init : int
582            The number of initial degrees of freedom in the unconstrained system.
583            This value is used to size the initial constraint matrix and manage
584            later extensions with reference DOFs.
585        """
586        self.nb_dofs_init = nb_dofs_init
587        self.lhs = sps.coo_matrix(np.empty((0, nb_dofs_init), dtype="float"))
588        self.rhs = np.empty(0, dtype="float")
589
590    def copy(self) -> "DirichletConstraintHandler":
591        """
592        Create a deep copy of this DirichletConstraintHandler instance.
593
594        Returns
595        -------
596        DirichletConstraintHandler
597            A new instance with the same initial number of DOFs, constraint matrix, and right-hand side vector.
598            All internal data is copied, so modifications to the returned handler do not affect the original.
599        """
600        return copy.deepcopy(self)

602    def add_eqs(self, lhs: sps.spmatrix, rhs: np.ndarray[np.floating]):
603        """
604        Append linear affine constraint equations to the global constraint system.
605
606        Adds equations of the form:
607
608            lhs @ u = rhs
609
610        to the accumulated constraint system `D @ u = c`.
611
612        If `lhs` is expressed only in terms of the initial physical DOFs
613        (`nb_dofs_init` columns), it is automatically extended with zero-padding
614        to match the current number of DOFs (e.g. after reference DOFs have been added).
615
616        Parameters
617        ----------
618        lhs : sps.spmatrix of shape (n_eqs, n_dofs)
619            Constraint matrix defining the left-hand side of the equations.
620            The number of columns must match either:
621            - the initial number of DOFs (`nb_dofs_init`), or
622            - the current number of DOFs in the handler.
623
624        rhs : np.ndarray[np.floating] of shape (n_eqs,)
625            Right-hand side values associated with the constraint equations.
626
627        Raises
628        ------
629        ValueError
630            If the number of columns of `lhs` is incompatible with the current
631            constraint system.
632
633        Notes
634        -----
635        Constraints are appended to the existing system and are not simplified
636        or checked for redundancy.
637        """
638        nb_eqs, nb_dofs = lhs.shape
639        assert nb_eqs == rhs.size
640        if nb_dofs == self.lhs.shape[1]:
641            self.lhs = sps.vstack((self.lhs, lhs))
642        elif nb_dofs == self.nb_dofs_init:
643            zero = sps.coo_matrix((nb_eqs, self.lhs.shape[1] - self.nb_dofs_init))
644            new_lhs = sps.hstack((lhs, zero))
645            self.lhs = sps.vstack((self.lhs, new_lhs))
646        else:
647            raise ValueError(
648                "lhs.shape[1] must match either the initial number of dofs or the current one."
649            )
650
651        self.rhs = np.hstack((self.rhs, rhs))
652
653    def add_ref_dofs(self, nb_dofs: int):
654        """
655        Append additional reference DOFs to the constraint system.
656
657        This method increases the size of the global DOF vector by adding
658        `nb_dofs` new reference DOFs. These DOFs are introduced without
659        imposing any constraint, i.e. they appear with zero coefficients
660        in all existing equations.
661
662        Parameters
663        ----------
664        nb_dofs : int
665            Number of reference DOFs to append to the global DOF vector.
666
667        Notes
668        -----
669        Reference DOFs are typically used to express kinematic relations,
670        rigid body motions, or other auxiliary constraint parametrizations.
671        """
672        zero = sps.coo_matrix((self.lhs.shape[0], nb_dofs))
673        self.lhs = sps.hstack((self.lhs, zero))
674
675    def add_ref_dofs_with_behavior(
676        self, behavior_mat: sps.spmatrix, slave_inds: np.ndarray[np.integer]
677    ):
678        """

682        relation linking these reference DOFs to existing DOFs (called *slave DOFs*).
683        The imposed constraints take the form:
684
685            behavior_mat @ ref_dofs - u[slave_inds] = 0
686

692            corresponding slave DOFs.

702            rows of `behavior_mat`.

707        appends the corresponding constraint equations.
708        """

710        assert nb_slaves == slave_inds.size
711        data = -np.ones(nb_slaves, dtype="float")
712        rows = np.arange(nb_slaves)
713        slaves_counterpart = sps.coo_matrix(
714            (data, (rows, slave_inds)), shape=(nb_slaves, self.lhs.shape[1])
715        )
716        lhs_to_add = sps.hstack((slaves_counterpart, behavior_mat))
717        rhs_to_add = np.zeros(nb_slaves, dtype="float")
718        self.add_ref_dofs(nb_ref_dofs)
719        self.add_eqs(lhs_to_add, rhs_to_add)

811    def make_C_k(self) -> tuple[sps.spmatrix, np.ndarray[np.floating]]:
812        """
813        Construct the affine transformation (C, k) enforcing all Dirichlet constraints.
814
815        This method solves the linear constraint system:
816
817            D @ u = c
818
819        by computing:
820
821        - a matrix `C` forming a basis of the nullspace of `D`
822        (i.e., D @ C = 0),
823        - a particular solution `k` such that D @ k = c.
824
825        Any admissible vector `u` satisfying the constraints can then be written as:
826
827            u = C @ dof + k
828
829        where `dof` is the reduced vector of unconstrained degrees of freedom.
830
831        The nullspace basis and the particular solution are obtained through
832        a sparse QR factorization of D.
833
834        Returns
835        -------
836        C : sps.spmatrix
837            Sparse matrix of shape (n_full_dofs, n_free_dofs) whose columns
838            form a basis of the nullspace of `D`.
839
840        k : np.ndarray[np.floating]
841            Vector of shape (n_full_dofs,) representing a particular solution
842            satisfying the constraints.
843
844        Notes
845        -----
846        - The transformation (C, k) defines an affine parametrization of the
847        admissible displacement space.
848        - The reduced DOF vector `dof` corresponds to the coordinates of `u`
849        in the nullspace basis.
850        - The QR factorization is performed on D to efficiently extract both
851        the nullspace basis and the particular solution.
852        """
853        # Step 1: Perform sparse QR factorization of D^T
854        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
855        Q, R, perm, rank = qr_sparse(self.lhs.T)
856        Q = Q.tocsc()
857        R = R.tocsc()
858
859        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
860        # These correspond to the last n - rank columns of Q
861        C = Q[:, rank:]
862
863        # Step 3: Compute a particular solution k such that D @ k = c
864        # (c = self.rhs contains the prescribed values)
865
866        # 3.1: Apply the column permutation to the RHS vector
867        c_tilde = self.rhs[np.array(perm)]
868
869        # 3.2: Extract the leading square block R1 of R (size rank  rank)
870        R1 = R[:rank, :rank].tocsc()
871
872        # 3.3: Take the first 'rank' entries of the permuted RHS
873        c1 = c_tilde[:rank]
874
875        # 3.4: Solve the triangular system R1^T @ y1 = c1
876        y1 = sps.linalg.spsolve(R1.T, c1)
877
878        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
879        y = np.zeros(Q.shape[0])
880        y[:rank] = y1
881
882        # 3.6: Recover the particular solution k = Q @ y
883        k = Q @ y
884
885        # Step 4: Return the nullspace basis C and the particular solution k
886        return C, k

888    def get_reduced_Ck(self) -> tuple[sps.spmatrix, np.ndarray]:
889        """
890        Extract the affine constraint transformation restricted to physical DOFs.
891
892        This method computes the full transformation (C, k) using `self.make_C_k()`
893        and returns only the rows associated with the initial (physical) degrees
894        of freedom. The resulting pair (C_u, k_u) defines:
895
896            u_phys = C_u @ dof + k_u
897
898        where `u_phys` satisfies all imposed Dirichlet constraints.
899
900        Returns
901        -------
902        C_u : sps.spmatrix
903            Reduced nullspace basis matrix of shape (nb_dofs_init, n_free_dofs).
904
905        k_u : np.ndarray
906            Reduced particular solution vector of shape (nb_dofs_init,).
907
908        Notes
909        -----
910        - The full transformation (C, k) may include auxiliary reference DOFs
911        introduced by multi-point or hierarchical constraints.
912        - Only the rows corresponding to physical DOFs are returned.
913        - A warning is emitted if reference DOFs remain unconstrained,
914        which may indicate missing boundary specifications.
915        """
916        C_ext, k_ext = self.make_C_k()

919        C_u = C_ext[: self.nb_dofs_init, :]
920        k_u = k_ext[: self.nb_dofs_init]
921        if sps.linalg.norm(C_ref) != 0:
922            print("Warning : not all reference point DoFs are specified.")
923            print(
924                f"Try specifying reference dofs number {np.abs(C_ref).sum(axis=1).A.ravel().nonzero()[0].tolist()}."
925            )

945    def get_ref_multipliers_from_internal_residual(self, K_u_minus_f):
946        """
947        Recover Lagrange multipliers associated with reference DOF constraints.
948
949        This method reconstructs the Lagrange multipliers  corresponding to
950        reference DOFs using the internal residual vector of the mechanical system:
951
952            r = K @ u - f
953
954        The derivation relies on the partition of the transformation matrix:
955
956            C = [C_u;
957                C_ref]
958
959        where `C_u` maps reduced DOFs to physical DOFs and `C_ref` maps them
960        to reference DOFs.
961
962        The multipliers satisfy:
963
964            C_ref.T @  = - C_u.T @ r
965
966        and are obtained via the least-squares solution:
967
968             = - (C_ref @ C_ref.T) @ C_ref @ C_u.T @ r
969
970        Parameters
971        ----------
972        K_u_minus_f : np.ndarray
973            Internal residual vector of shape (nb_dofs_init,),
974            corresponding to K @ u - f restricted to physical DOFs.
975
976        Returns
977        -------
978        lamb : np.ndarray
979            Vector of Lagrange multipliers associated with reference DOF constraints.
980
981        Notes
982        -----
983        - The multipliers correspond to reaction forces or generalized constraint
984        forces transmitted through reference DOFs.
985        - The residual must be assembled consistently with the stiffness matrix
986        and load vector used to compute the displacement field.
987        - The matrix C_ref @ C_ref.T is assumed to be invertible, which holds
988        when reference constraints are linearly independent.
989        """
990        C, _ = self.make_C_k()
991        C_ref = C[self.nb_dofs_init :, :]
992        C_u = C[: self.nb_dofs_init, :]
993        lamb = -solve_sparse(C_ref @ C_ref.T, C_ref @ C_u.T @ K_u_minus_f)
994        return lamb

128        Impose pointwise Dirichlet conditions by prescribing displacement values
129        at selected full DOFs.
130
131        This method enforces constraints of the form:
132
133            u[i] = vals[j]     for i = inds[j]
134
135        by modifying the affine mapping:
136
137            u = C @ dof + k
138
139        The modification removes the corresponding admissible variations in `C`
140        and updates `k` so that the prescribed values are always satisfied.
141        As a result, constrained DOFs are eliminated from the reduced DOF vector.
142
143        Parameters

145        inds : np.ndarray[np.integer]

154        - Columns of `C` that become unused are removed, which may reduce
155          the number of reduced DOFs.
156        - The affine offset `k` is updated to enforce the prescribed values.
157        - This operation modifies the current Dirichlet object in place.

160        def zero_rows(C: sps.coo_matrix, rows: np.ndarray[np.integer]):
161            rm = np.hstack([(C.row == row).nonzero()[0] for row in rows])
162            row = np.delete(C.row, rm)
163            col = np.delete(C.col, rm)
164            data = np.delete(C.data, rm)

179    def slave_reference_linear_relation(
180        self,
181        slaves: np.ndarray[int],
182        references: np.ndarray[int],
183        coefs: Union[np.ndarray[float], None] = None,
184    ):

186        Impose linear multi-point constraints between full DOFs.
187
188        This method enforces relations of the form:
189
190            u[slave_i] = sum_j coefs[i, j] * u[references[i, j]]
191
192        by modifying the affine mapping:
193
194            u = C @ dof + k
195
196        Slave DOFs are expressed as linear combinations of reference DOFs.
197        The corresponding admissible variations are propagated into `C`
198        and `k`, effectively eliminating slave DOFs from the reduced parameter
199        vector while preserving the imposed constraints.
200

203        slaves : np.ndarray[int] of shape (n_slaves,)
204            Indices of DOFs that are constrained (slave DOFs).
205        references : np.ndarray[int] of shape (n_slaves, n_refs)

208        coefs : np.ndarray[float], optional, shape (n_slaves, n_refs)
209            Linear combination coefficients linking references to slaves.
210            If None, slaves are defined as the average of their references.
211
212        Notes
213        -----
214        - Slave DOFs are removed from the reduced DOF vector.
215        - The constraint propagation accounts for hierarchical dependencies
216          between slave DOFs using a topological ordering.
217        - This operation is typically faster than using `DirichletConstraintHandler`,
218          as it directly modifies the affine mapping without constructing
219          intermediate constraint objects.
220        - Cyclic dependencies between slaves are not supported and will raise an error.
221        - This operation modifies the current Dirichlet object in place.
222
223        Examples
224        --------
225        To impose u[i] = 0.5 * (u[j] + u[k]):
226
227        >>> slaves = np.array([i])
228        >>> references = np.array([[j, k]])

230        >>> dirichlet.slave_reference_linear_relation(slaves, references, coefs)

244            slaves,
245            references,
246            coefs,
247            sorted_slaves,
248        )
249        C = sps.coo_matrix((data, (rows, cols)), shape=C.shape)
250
251        unique, inverse = np.unique(C.col, return_inverse=True)
252        C.col = np.arange(unique.size)[inverse]
253        C._shape = (C.shape[0], unique.size)
254        self.C = C.tocsc()

327    def dof_lsq(self, u: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
328        """
329        Compute reduced DOFs from a displacement field using a least-squares projection.
330
331        This method computes `dof` such that:
332
333            u  C @ dof + k
334
335        by solving the normal equations:
336
337            (C C) dof = C (u - k)
338
339        Parameters
340        ----------
341        u : np.ndarray[np.floating]
342            Displacement field. Can be either:
343
344            - shape (n_u,)
345            - shape (..., n_u) for batched projection
346
347        Returns
348        -------
349        dof : np.ndarray[np.floating]
350            Reduced degrees of freedom with shape:
351
352            - (n_dof,) if `u` is 1D
353            - (..., n_dof) if `u` is batched
354
355        Notes
356        -----
357        This operation performs a least-squares inversion of the affine mapping.
358        If `C` does not have full column rank, the solution corresponds to a
359        minimum-norm least-squares solution.
360
361        The system `(C C)` is solved using a sparse linear solver.
362        """
363        if u.ndim == 1:
364            dof = solve_sparse(self.C.T @ self.C, self.C.T @ (u - self.k))
365        else:
366            m, n = self.C.shape
367            u_shape = u.shape[:-1]
368            dof = solve_sparse(
369                self.C.T @ self.C, (u.reshape((-1, m)) - self.k[None]) @ self.C
370            ).reshape((*u_shape, n))

812    def make_C_k(self) -> tuple[sps.spmatrix, np.ndarray[np.floating]]:
813        """
814        Construct the affine transformation (C, k) enforcing all Dirichlet constraints.
815
816        This method solves the linear constraint system:
817
818            D @ u = c
819
820        by computing:
821
822        - a matrix `C` forming a basis of the nullspace of `D`
823        (i.e., D @ C = 0),
824        - a particular solution `k` such that D @ k = c.
825
826        Any admissible vector `u` satisfying the constraints can then be written as:
827
828            u = C @ dof + k
829
830        where `dof` is the reduced vector of unconstrained degrees of freedom.
831
832        The nullspace basis and the particular solution are obtained through
833        a sparse QR factorization of D.
834
835        Returns
836        -------
837        C : sps.spmatrix
838            Sparse matrix of shape (n_full_dofs, n_free_dofs) whose columns
839            form a basis of the nullspace of `D`.
840
841        k : np.ndarray[np.floating]
842            Vector of shape (n_full_dofs,) representing a particular solution
843            satisfying the constraints.
844
845        Notes
846        -----
847        - The transformation (C, k) defines an affine parametrization of the
848        admissible displacement space.
849        - The reduced DOF vector `dof` corresponds to the coordinates of `u`
850        in the nullspace basis.
851        - The QR factorization is performed on D to efficiently extract both
852        the nullspace basis and the particular solution.
853        """
854        # Step 1: Perform sparse QR factorization of D^T
855        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
856        Q, R, perm, rank = qr_sparse(self.lhs.T)
857        Q = Q.tocsc()
858        R = R.tocsc()
859
860        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
861        # These correspond to the last n - rank columns of Q
862        C = Q[:, rank:]
863
864        # Step 3: Compute a particular solution k such that D @ k = c
865        # (c = self.rhs contains the prescribed values)
866
867        # 3.1: Apply the column permutation to the RHS vector
868        c_tilde = self.rhs[np.array(perm)]
869
870        # 3.2: Extract the leading square block R1 of R (size rank  rank)
871        R1 = R[:rank, :rank].tocsc()
872
873        # 3.3: Take the first 'rank' entries of the permuted RHS
874        c1 = c_tilde[:rank]
875
876        # 3.4: Solve the triangular system R1^T @ y1 = c1
877        y1 = sps.linalg.spsolve(R1.T, c1)
878
879        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
880        y = np.zeros(Q.shape[0])
881        y[:rank] = y1
882
883        # 3.6: Recover the particular solution k = Q @ y
884        k = Q @ y
885
886        # Step 4: Return the nullspace basis C and the particular solution k
887        return C, k

def make_C_k( self) -> tuple[scipy.sparse._matrix.spmatrix, numpy.ndarray[numpy.floating]]: 
812    def make_C_k(self) -> tuple[sps.spmatrix, np.ndarray[np.floating]]:
813        """
814        Construct the affine transformation (C, k) enforcing all Dirichlet constraints.
815
816        This method solves the linear constraint system:
817
818            D @ u = c
819
820        by computing:
821
822        - a matrix `C` forming a basis of the nullspace of `D`
823        (i.e., D @ C = 0),
824        - a particular solution `k` such that D @ k = c.
825
826        Any admissible vector `u` satisfying the constraints can then be written as:
827
828            u = C @ dof + k
829
830        where `dof` is the reduced vector of unconstrained degrees of freedom.
831
832        The nullspace basis and the particular solution are obtained through
833        a sparse QR factorization of D.
834
835        Returns
836        -------
837        C : sps.spmatrix
838            Sparse matrix of shape (n_full_dofs, n_free_dofs) whose columns
839            form a basis of the nullspace of `D`.
840
841        k : np.ndarray[np.floating]
842            Vector of shape (n_full_dofs,) representing a particular solution
843            satisfying the constraints.
844
845        Notes
846        -----
847        - The transformation (C, k) defines an affine parametrization of the
848        admissible displacement space.
849        - The reduced DOF vector `dof` corresponds to the coordinates of `u`
850        in the nullspace basis.
851        - The QR factorization is performed on D to efficiently extract both
852        the nullspace basis and the particular solution.
853        """
854        # Step 1: Perform sparse QR factorization of D^T
855        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
856        Q, R, perm, rank = qr_sparse(self.lhs.T)
857        Q = Q.tocsc()
858        R = R.tocsc()
859
860        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
861        # These correspond to the last n - rank columns of Q
862        C = Q[:, rank:]
863
864        # Step 3: Compute a particular solution k such that D @ k = c
865        # (c = self.rhs contains the prescribed values)
866
867        # 3.1: Apply the column permutation to the RHS vector
868        c_tilde = self.rhs[np.array(perm)]
869
870        # 3.2: Extract the leading square block R1 of R (size rank  rank)
871        R1 = R[:rank, :rank].tocsc()
872
873        # 3.3: Take the first 'rank' entries of the permuted RHS
874        c1 = c_tilde[:rank]
875
876        # 3.4: Solve the triangular system R1^T @ y1 = c1
877        y1 = sps.linalg.spsolve(R1.T, c1)
878
879        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
880        y = np.zeros(Q.shape[0])
881        y[:rank] = y1
882
883        # 3.6: Recover the particular solution k = Q @ y
884        k = Q @ y
885
886        # Step 4: Return the nullspace basis C and the particular solution k
887        return C, k
