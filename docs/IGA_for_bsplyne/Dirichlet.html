 10class Dirichlet:
 11    """
 12    A class to handle Dirichlet boundary conditions (BC) for a problem using an affine mapping.
 13
 14    The Dirichlet class provides methods to apply Dirichlet BCs by mapping degrees of freedom (dof)
 15    to displacements (u) using the relation `u = C @ dof + k`. It supports creating instances with
 16    identity mappings, locking specific displacement indices, and computing dof from displacements
 17    via least squares approximation.
 18
 19    Attributes
 20    ----------
 21    C : sps.csc_matrix
 22        The matrix used in the affine mapping from `dof` to `u`.
 23    k : np.ndarray[np.floating]
 24        The vector used in the affine mapping from `dof` to `u`.
 25    """
 26
 27    C: sps.csc_matrix
 28    k: np.ndarray[np.floating]
 29
 30    def __init__(self, C: sps.spmatrix, k: np.ndarray[np.floating]):
 31        """
 32        Initializes a Dirichlet instance with the given matrix and vector for affine mapping.
 33
 34        Parameters
 35        ----------
 36        C : sps.spmatrix of float
 37            The matrix used to map degrees of freedom `dof` to displacements `u`.
 38        k : np.ndarray[np.floating]
 39            The vector used to map degrees of freedom `dof` to displacements `u`.
 40        """
 41        self.C = C.tocsc()
 42        self.k = k

 44    @classmethod
 45    def eye(cls, size: int):
 46        """
 47        Create a `Dirichlet` instance with an identity mapping, where no degrees of freedom `dof` are locked.
 48        Sets `C` to the identity matrix and `k` to a zero-filled vector.
 49
 50        Parameters
 51        ----------
 52        size : int
 53            Size of the `dof` and `u` vectors.
 54
 55        Returns
 56        -------
 57        dirichlet : Dirichlet
 58            The identity `Dirichlet` instance.
 59        """
 60        C = sps.eye(size)
 61        k = np.zeros(size, dtype="float")
 62        return cls(C, k)

 64    @classmethod
 65    def lock_disp_inds(cls, inds: np.ndarray[np.integer], k: np.ndarray[np.floating]):
 66        """
 67        Creates a `Dirichlet` instance with specified displacement `u` indices locked to given values.
 68
 69        Parameters
 70        ----------
 71        inds : np.ndarray[np.integer]
 72            Indices of the displacement field `u` to be locked.
 73        k : np.ndarray[np.floating]
 74            Values to lock the specified indices of `u` to.
 75            After this, `u[inds]` are set to `k[inds]` while other compontents are left free.
 76
 77        Returns
 78        -------
 79        dirichlet : Dirichlet
 80            A `Dirichlet` instance with specified displacements locked.
 81        """
 82        C = sps.eye(k.size)
 83        dirichlet = cls(C, k.copy())
 84        dirichlet.set_u_inds_vals(inds, dirichlet.k[inds])
 85        return dirichlet

 87    def set_u_inds_vals(
 88        self, inds: np.ndarray[np.integer], vals: np.ndarray[np.floating]
 89    ):
 90        """
 91        Locks specified indices of the displacement field `u` to given values by modifying
 92        the matrix `C` and vector `k` accordingly. This involves zeroing out the specified
 93        rows in `C` and adjusting `k` to reflect the locked values.
 94
 95        Parameters
 96        ----------
 97        inds : np.ndarray[np.integer]
 98            Indices of the displacement field `u` to be locked.
 99        vals : np.ndarray[np.floating]
100            Values to lock the specified indices of `u` to.
101        """
102
103        def zero_rows(C: sps.coo_matrix, rows: np.ndarray[np.integer]):
104            rm = np.hstack([(C.row == row).nonzero()[0] for row in rows])
105            row = np.delete(C.row, rm)
106            col = np.delete(C.col, rm)
107            data = np.delete(C.data, rm)
108            C = sps.coo_matrix((data, (row, col)), shape=(C.shape))
109            return C
110
111        def remove_zero_cols(C: sps.coo_matrix):
112            unique, inverse = np.unique(C.col, return_inverse=True)
113            C.col = np.arange(unique.size)[inverse]
114            C._shape = (C.shape[0], unique.size)
115
116        if inds.size != 0:
117            C_coo = zero_rows(self.C.tocoo(), inds)
118            remove_zero_cols(C_coo)
119            self.C = C_coo.tocsc()
120            self.k[inds] = vals

122    def slave_reference_linear_relation(
123        self,
124        slaves: np.ndarray[int],
125        references: np.ndarray[int],
126        coefs: Union[np.ndarray[float], None] = None,
127    ):
128        """
129        This function modifies the sparse matrix `C` and the vector `k` to enforce
130        reference-slave constraints in an optimization problem. The goal is to eliminate
131        the degrees of freedom (DOFs) associated with slave nodes, keeping only the reference
132        DOFs. The relation `u = C@dof + k` is updated so that slave DOFs are expressed as
133        linear combinations of reference DOFs, reducing the problem's size while maintaining
134        the imposed constraints.
135
136        Parameters
137        ----------
138        slaves : np.ndarray[int]
139            Array of slave indices.
140        references : np.ndarray[int]
141            2D array where each row contains the reference indices controlling a slave.
142        coefs : Union[np.ndarray[float], None], optional
143            2D array of coefficients defining the linear relationship between references and
144            slaves.
145            If None, the coefficients are set so that the slaves are the average of the references.
146            By default None.
147        """
148        if coefs is None:
149            coefs = np.ones(references.shape, dtype="float") / references.shape[1]
150
151        sorted_slaves = slave_reference_linear_relation_sort(slaves, references)
152
153        C = self.C.tocsr()
154        rows, cols, data, self.k = slave_reference_linear_relation_inner(
155            C.indices,
156            C.indptr,
157            C.data,
158            self.k,
159            slaves,
160            references,
161            coefs,
162            sorted_slaves,
163        )
164        C = sps.coo_matrix((data, (rows, cols)), shape=C.shape)
165
166        unique, inverse = np.unique(C.col, return_inverse=True)
167        C.col = np.arange(unique.size)[inverse]
168        C._shape = (C.shape[0], unique.size)
169        self.C = C.tocsc()

171    def u_du_ddof(
172        self, dof: np.ndarray[np.floating]
173    ) -> tuple[np.ndarray[np.floating], sps.csc_matrix]:
174        """
175        Computes the displacement field `u` and its derivative with respect to the degrees of freedom `dof`.
176        The displacement field is calculated as `u = C @ dof + k`, and its derivative is `C`.
177
178        Parameters
179        ----------
180        dof : np.ndarray[np.floating]
181            The degrees of freedom of the problem, representing the input to the affine mapping.
182
183        Returns
184        -------
185        u, du_ddof : tuple[np.ndarray[np.floating], sps.csc_matrix]
186            A tuple containing the displacement field `u` and its derivative with respect to `dof`.
187        """
188        u = self.C @ dof + self.k
189        du_ddof = self.C
190        return u, du_ddof
191
192    def u(self, dof: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
193        """
194        Computes the displacement field `u` using the affine mapping `u = C @ dof + k`.
195
196        Parameters
197        ----------
198        dof : np.ndarray[np.floating]
199            The degrees of freedom of the problem, representing the input to the affine mapping.
200
201        Returns
202        -------
203        u : np.ndarray[np.floating]
204            The computed displacement field `u`.
205        """
206        if dof.ndim == 1:
207            u = self.C @ dof + self.k
208        else:
209            m, n = self.C.shape
210            dof_shape = dof.shape[:-1]
211            u = (self.C @ dof.reshape((-1, n)).T + self.k.reshape((m, 1))).T.reshape(
212                (*dof_shape, m)
213            )
214        return u
215
216    def dof_lsq(self, u: np.ndarray[np.floating]) -> np.ndarray[np.floating]:
217        """
218        Computes the degrees of freedom `dof` from the displacement field `u` using a least squares approximation.
219        This method performs a least squares 'inversion' of the affine mapping `u = C @ dof + k`.
220        It solves the linear problem `C.T @ C @ dof = C.T @ (u - k)` for `dof`.
221
222        Parameters
223        ----------
224        u : np.ndarray[np.floating]
225            The displacement field from which to compute the degrees of freedom.
226
227        Returns
228        -------
229        dof : np.ndarray[np.floating]
230            The computed degrees of freedom corresponding to the given displacement field.
231        """
232        if u.ndim == 1:
233            dof = cholesky(self.C.T @ self.C)(self.C.T @ (u - self.k))
234        else:
235            m, n = self.C.shape
236            u_shape = u.shape[:-1]
237            dof = cholesky(self.C.T @ self.C)(
238                (u.reshape((-1, m)) - self.k[None]) @ self.C
239            ).reshape((*u_shape, n))
240        return dof

243@nb.njit(cache=True)
244def slave_reference_linear_relation_sort(
245    slaves: np.ndarray[int], references: np.ndarray[int]
246) -> np.ndarray[int]:
247    """
248    Sorts the slave nodes based on reference indices to respect
249    hierarchical dependencies (each slave is processed after its references).
250
251    Parameters
252    ----------
253    slaves : np.ndarray[int]
254        Array of slave indices.
255    references : np.ndarray[int]
256        2D array where each row contains the reference indices controlling a slave.
257
258    Returns
259    -------
260    sorted_slaves : np.ndarray[int]
261        Array of slave indices sorted based on dependencies.
262    """
263    slaves_set = set(slaves)
264    graph = {s: nb.typed.List.empty_list(nb.int64) for s in slaves}
265    indegree = {s: 0 for s in slaves}
266
267    # Building the graph
268    for i in range(len(slaves)):
269        for reference in references[i]:
270            if reference in slaves_set:
271                indegree[reference] += 1
272                graph[slaves[i]].append(reference)
273
274    # Queue for slaves with no dependencies (in-degree 0)
275    queue = [s for s in slaves if indegree[s] == 0]
276
277    # Topological sorting via BFS
278    sorted_slaves = np.empty(len(slaves), dtype="int")
279    i = 0
280    while queue:
281        slave = queue.pop(0)
282        sorted_slaves[i] = slave
283        i += 1
284        for dependent_slave in graph[slave]:
285            indegree[dependent_slave] -= 1
286            if indegree[dependent_slave] == 0:
287                queue.append(dependent_slave)
288
289    if i != len(slaves):
290        raise ValueError("Cyclic dependency detected in slaves.")
291
292    sorted_slaves = sorted_slaves[::-1]
293    return sorted_slaves

296@nb.njit(cache=True)
297def slave_reference_linear_relation_inner(
298    indices: np.ndarray[int],
299    indptr: np.ndarray[int],
300    data: np.ndarray[float],
301    k: np.ndarray[float],
302    slaves: np.ndarray[int],
303    references: np.ndarray[int],
304    coefs: np.ndarray[float],
305    sorted_slaves: np.ndarray[int],
306) -> tuple[np.ndarray[int], np.ndarray[int], np.ndarray[float], np.ndarray[float]]:
307    """
308    Applies slave-reference relations directly to CSR matrix arrays.
309
310    Parameters
311    ----------
312    indices : np.ndarray[int]
313        Column indices of CSR matrix.
314    indptr : np.ndarray[int]
315        Row pointers of CSR matrix.
316    data : np.ndarray[float]
317        Nonzero values of CSR matrix.
318    k : np.ndarray[float]
319        Vector to be updated.
320    slaves : np.ndarray[int]
321        Array of slave indices.
322    references : np.ndarray[int]
323        2D array where each row contains the reference indices controlling a slave.
324    coefs : np.ndarray[float]
325        2D array of coefficients defining the linear relationship between references and
326        slaves.
327    sorted_slaves : np.ndarray[int]
328        Array of slave indices sorted in topological order.
329
330    Returns
331    -------
332    rows : np.ndarray[int]
333        Updated row indices of COO matrix.
334    cols : np.ndarray[int]
335        Updated column indices of COO matrix.
336    data : np.ndarray[float]
337        Updated nonzero values of COO matrix.
338    k : np.ndarray[float]
339        Updated vector.
340    """
341
342    # Convert CSR to list of dicts
343    dict_list = [
344        {indices[j]: data[j] for j in range(indptr[i], indptr[i + 1])}
345        for i in range(indptr.size - 1)
346    ]
347
348    # Apply linear relation
349    slaves_to_index = {s: i for i, s in enumerate(slaves)}
350    for slave in sorted_slaves:
351        i = slaves_to_index[slave]
352        new_row = {}
353        for reference_ind, coef in zip(references[i], coefs[i]):
354            reference_row = dict_list[reference_ind]
355            for ind, val in reference_row.items():
356                if ind in new_row:
357                    new_row[ind] += coef * val
358                else:
359                    new_row[ind] = coef * val
360        new_row = {ind: val for ind, val in new_row.items() if val != 0}
361        dict_list[slave] = new_row
362        k[slave] = np.dot(k[references[i]], coefs[i])
363
364    # Convert list of dicts to COO
365    nnz = sum([len(row) for row in dict_list])
366    rows = np.empty(nnz, dtype=np.int32)
367    cols = np.empty(nnz, dtype=np.int32)
368    data = np.empty(nnz, dtype=np.float64)
369    pos = 0
370    for i, row in enumerate(dict_list):
371        for j, val in row.items():
372            rows[pos] = i
373            cols[pos] = j
374            data[pos] = val
375            pos += 1
376
377    return rows, cols, data, k

381class DirichletConstraintHandler:
382    """
383    Manage linear Dirichlet constraints for variational or optimization problems.
384
385    This class accumulates linear equations of the form `D @ u = c` representing Dirichlet
386    boundary conditions or linear relationships between degrees of freedom (DOFs), and
387    computes a reduced basis representation that parametrizes the set of admissible solutions.
388
389    Specifically, it computes matrices `C` and `k` such that any vector `u` satisfying
390    `D @ u = c` can be written as:
391
392        u = C @ dof + k
393
394    where `dof` is a reduced vector of free parameters.
395
396    Attributes
397    ----------
398    nb_dofs_init : int
399        Number of DOFs in the original unconstrained system, before adding any reference DOFs.
400    lhs : sps.spmatrix
401        Accumulated constraint matrix D (left-hand side of the Dirichlet conditions).
402    rhs : np.ndarray[np.floating]
403        Accumulated right-hand side vector c of the Dirichlet conditions.
404    """
405
406    nb_dofs_init: int
407    lhs: sps.spmatrix
408    rhs: np.ndarray[np.floating]
409
410    def __init__(self, nb_dofs_init: int):
411        """
412        Initialize a Dirichlet constraint handler.
413
414        Parameters
415        ----------
416        nb_dofs_init : int
417            The number of initial degrees of freedom in the unconstrained system.
418            This value is used to size the initial constraint matrix and manage
419            later extensions with reference DOFs.
420        """
421        try:
422            from sparseqr import qr
423
424            self._qr = qr
425        except ImportError as e:
426            raise ModuleNotFoundError(
427                "The package 'sparseqr' is required for QR decomposition but is not installed.\n"
428                "\n"
429                "Note: Installing 'sparseqr' may require a C++ compiler, CMake, and the SuiteSparse library.\n"
430                "On Linux, you can typically install these dependencies via your package manager. For example:\n"
431                "    sudo apt install libsuitesparse-dev cmake build-essential\n"
432                "\n"
433                "Then install sparseqr with pip:\n"
434                "    pip install sparseqr\n"
435                "\n"
436                "If installation continues to fail, refer to the official documentation:\n"
437                "    https://pypi.org/project/sparseqr/"
438            ) from e
439        self.nb_dofs_init = nb_dofs_init
440        self.lhs = sps.coo_matrix(np.empty((0, nb_dofs_init), dtype="float"))
441        self.rhs = np.empty(0, dtype="float")
442
443    def copy(self) -> "DirichletConstraintHandler":
444        """
445        Create a deep copy of this DirichletConstraintHandler instance.
446
447        Returns
448        -------
449        DirichletConstraintHandler
450            A new instance with the same initial number of DOFs, constraint matrix, and right-hand side vector.
451            All internal data is copied, so modifications to the returned handler do not affect the original.
452        """
453        return copy.deepcopy(self)

455    def add_eqs(self, lhs: sps.spmatrix, rhs: np.ndarray[np.floating]):
456        """
457        Add linear constraint equations of the form D_new @ u = c_new.
458
459        Appends the given equations to the existing Dirichlet constraint system.
460        If the number of columns in `lhs` matches the initial DOF count, it is
461        automatically extended with zero-padding to match the current DOF count
462        (in case reference DOFs have been added).
463
464        Parameters
465        ----------
466        lhs : sps.spmatrix
467            Constraint matrix D_new of shape (n_eqs, nb_dofs) to be added to the system.
468        rhs : np.ndarray[np.floating]
469            Right-hand side values c_new of shape (n_eqs,) corresponding to the constraint.
470
471        Raises
472        ------
473        ValueError
474            If the number of columns in `lhs` does not match the initial or current DOF count.
475        """
476        nb_eqs, nb_dofs = lhs.shape
477        assert nb_eqs == rhs.size
478        if nb_dofs == self.lhs.shape[1]:
479            self.lhs = sps.vstack((self.lhs, lhs))
480        elif nb_dofs == self.nb_dofs_init:
481            zero = sps.coo_matrix((nb_eqs, self.lhs.shape[1] - self.nb_dofs_init))
482            new_lhs = sps.hstack((lhs, zero))
483            self.lhs = sps.vstack((self.lhs, new_lhs))
484        else:
485            raise ValueError(
486                "lhs.shape[1] must match either the initial number of dofs or the current one."
487            )
488
489        self.rhs = np.hstack((self.rhs, rhs))
490
491    def add_ref_dofs(self, nb_dofs: int):
492        """
493        Extend the system by adding new reference DOFs.
494
495        This increases the number of columns in the constraint matrix by `nb_dofs`,
496        initializing them with zeros in all existing constraint equations.
497
498        Parameters
499        ----------
500        nb_dofs : int
501            Number of new reference DOFs to append at the end of the current DOF vector.
502        """
503        zero = sps.coo_matrix((self.lhs.shape[0], nb_dofs))
504        self.lhs = sps.hstack((self.lhs, zero))

506    def add_ref_dofs_with_behavior(
507        self, behavior_mat: sps.spmatrix, slave_inds: np.ndarray[np.integer]
508    ):
509        """
510        Add new reference DOFs and define their influence on existing DOFs via a behavioral relation.
511
512        This method appends new reference DOFs and enforces their relationship to existing
513        DOFs (called "slaves") through a linear behavior matrix `behavior_mat`. The resulting constraints
514        take the form:
515
516            `behavior_mat @ ref_dofs - u[slave_inds] = 0`
517
518        and are added to the global constraint system.
519
520        Parameters
521        ----------
522        behavior_mat : sps.spmatrix
523            Matrix of shape `(n_slaves, n_ref_dofs)` defining how each reference DOF influences
524            the corresponding slave DOFs.
525
526        slave_inds : np.ndarray[np.integer]
527            Indices of the slave DOFs. Must have length `n_slaves` and must be in the same
528            order as the rows of `behavior_mat`.
529
530        Raises
531        ------
532        AssertionError
533            If the number of rows in `behavior_mat` does not match the size of `slave_inds`.
534        """
535        nb_slaves, nb_ref_dofs = behavior_mat.shape
536        assert nb_slaves == slave_inds.size
537        data = -np.ones(nb_slaves, dtype="float")
538        rows = np.arange(nb_slaves)
539        slaves_counterpart = sps.coo_matrix(
540            (data, (rows, slave_inds)), shape=(nb_slaves, self.lhs.shape[1])
541        )
542        lhs_to_add = sps.hstack((slaves_counterpart, behavior_mat))
543        rhs_to_add = np.zeros(nb_slaves, dtype="float")
544        self.add_ref_dofs(nb_ref_dofs)
545        self.add_eqs(lhs_to_add, rhs_to_add)

547    def add_rigid_body_constraint(
548        self,
549        ref_point: np.ndarray[np.floating],
550        slaves_inds: np.ndarray[np.integer],
551        slaves_positions: np.ndarray[np.floating],
552    ):
553        """
554        Add a reference node and impose a rigid body motion constraint on a set of slave nodes.
555
556        This method introduces new reference degrees of freedom (DOFs) corresponding to a reference node
557        located at `ref_point`, and constrains the displacements of a set of slave nodes (given by
558        `nodes_inds` and `nodes_positions`) to follow a rigid body motion defined by the reference node.
559        The rigid body motion includes both translation and rotation about the reference point.
560
561        The imposed constraint ensures that the displacement of each slave node is a linear combination of
562        the reference node's translation and rotation, enforcing a rigid connection between the reference
563        and the slaves.
564
565        Parameters
566        ----------
567        ref_point : np.ndarray[np.floating]
568            Reference point (origin for rotation and translation), array of shape (3,) representing (x, y, z).
569        slaves_inds : np.ndarray[np.integer]
570            Indices of the degrees of freedom of the slave displacements.
571            Shape (3, n), where n is the number of slave nodes; each column contains the x, y, z DOF indices for a node.
572        slaves_positions : np.ndarray[np.floating]
573            Initial positions of the slave nodes.
574            Shape (3, n), where each column contains the (x, y, z) coordinates of a slave node in the physical space.
575        """
576        x, y, z = (
577            slaves_positions[:, :, None] - ref_point[:, None, None]
578        )  # x is of shape (n, 1)
579        ones = np.ones_like(x)
580        behavior = sps.bmat(
581            [
582                [None, z, -y, ones, None, None],
583                [-z, None, x, None, ones, None],
584                [y, -x, None, None, None, ones],
585            ]
586        )  # behavior matrix for U(X; theta, t) = theta \carat X + t
587        inds = slaves_inds.ravel()
588        self.add_ref_dofs_with_behavior(behavior, inds)

590    def add_eqs_from_inds_vals(
591        self, inds: np.ndarray[np.integer], vals: np.ndarray[np.floating] = None
592    ):
593        """
594        Add pointwise Dirichlet conditions by prescribing values at specific DOFs.
595
596        This is a convenience method for adding equations of the form:
597
598            u[i] = v
599
600        for given indices `i` and corresponding values `v`.
601
602        Parameters
603        ----------
604        inds : np.ndarray[np.integer]
605            Indices of the DOFs to constrain.
606
607        vals : np.ndarray[np.floating], optional
608            Values to prescribe at the corresponding indices. If None, zeros are used.
609            Must have the same size as `inds`.
610
611        Raises
612        ------
613        AssertionError
614            If `vals` is provided and does not match the shape of `inds`.
615        """
616        nb_eqs = inds.size
617        vals = np.zeros(nb_eqs, dtype="float") if vals is None else vals

624    def make_C_k(self) -> tuple[sps.spmatrix, np.ndarray[np.floating]]:
625        """
626        Construct the transformation (C, k) that enforces the Dirichlet constraints.
627
628        Solves the linear constraint system D @ u = c by computing:
629        - a basis `C` for the nullspace of D (i.e., D @ C = 0),
630        - a particular solution `k` such that D @ k = c.
631
632        Any admissible vector u satisfying the constraints can then be written as:
633
634            u = C @ dof + k
635
636        where `dof` is a reduced set of unconstrained degrees of freedom.
637
638        Returns
639        -------
640        C : sps.spmatrix
641            Basis of the nullspace of D, of shape (n_full_dofs, n_free_dofs), such that D @ C = 0.
642
643        k : np.ndarray[np.floating]
644            Particular solution of shape (n_full_dofs,) such that D @ k = c.
645        """
646        # Step 1: Perform sparse QR factorization of D^T
647        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
648        Q, R, perm, rank = self._qr(self.lhs.T, economy=False)
649        Q = Q.tocsc()
650        R = R.tocsc()
651
652        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
653        # These correspond to the last n - rank columns of Q
654        C = Q[:, rank:]
655
656        # Step 3: Compute a particular solution k such that D @ k = c
657        # (c = self.rhs contains the prescribed values)
658
659        # 3.1: Apply the column permutation to the RHS vector
660        c_tilde = self.rhs[np.array(perm)]
661
662        # 3.2: Extract the leading square block R1 of R (size rank  rank)
663        R1 = R[:rank, :rank].tocsc()
664
665        # 3.3: Take the first 'rank' entries of the permuted RHS
666        c1 = c_tilde[:rank]
667
668        # 3.4: Solve the triangular system R1^T @ y1 = c1
669        y1 = sps.linalg.spsolve(R1.T, c1)
670
671        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
672        y = np.zeros(Q.shape[0])
673        y[:rank] = y1
674
675        # 3.6: Recover the particular solution k = Q @ y
676        k = Q @ y
677
678        # Step 4: Return the nullspace basis C and the particular solution k
679        return C, k

682        """
683        Compute and return the reduced transformation matrices `(C_u, k_u)` for the physical degrees of freedom.
684
685        This method extracts only the first `nb_dofs_init` rows of the full constraint transformation matrices `(C, k)`
686        produced by `make_C_k()`, yielding a pair `(C_u, k_u)` such that the physical solution vector
687        `u_phys = C_u @ dof_ext + k_u` satisfies all imposed Dirichlet constraints for the original system.

689        Returns
690        -------
691        tuple[sps.spmatrix, np.ndarray]
692            A tuple `(C_u, k_u)` where:
693            - `C_u` (`sps.spmatrix`): The reduced nullspace basis matrix of shape (`nb_dofs_init`, n_free_dofs).
694            - `k_u` (`np.ndarray`): The reduced particular solution vector of shape (`nb_dofs_init`,).

698        - Only the rows corresponding to the initial (physical) degrees of freedom are returned.
699        - The full transformation matrices may include additional reference DOFs, which are omitted here.
700        """
701        C_ext, k_ext = self.make_C_k()

718        """
719        Compute the Lagrange multipliers associated with reference point constraints
720        from the internal residual vector of the mechanical problem.
721
722        This method reconstructs the multipliers  enforcing the constraints linked to
723        reference degrees of freedom (DOFs) using the internal residual
724        `K_u_minus_f = K @ u - f`. The derivation relies on the relation:
725

727

759        lamb = -cholesky(C_ref @ C_ref.T)(C_ref @ C_u.T @ K_u_minus_f)

625    def make_C_k(self) -> tuple[sps.spmatrix, np.ndarray[np.floating]]:
626        """
627        Construct the transformation (C, k) that enforces the Dirichlet constraints.
628
629        Solves the linear constraint system D @ u = c by computing:
630        - a basis `C` for the nullspace of D (i.e., D @ C = 0),
631        - a particular solution `k` such that D @ k = c.
632
633        Any admissible vector u satisfying the constraints can then be written as:
634
635            u = C @ dof + k
636
637        where `dof` is a reduced set of unconstrained degrees of freedom.
638
639        Returns
640        -------
641        C : sps.spmatrix
642            Basis of the nullspace of D, of shape (n_full_dofs, n_free_dofs), such that D @ C = 0.
643
644        k : np.ndarray[np.floating]
645            Particular solution of shape (n_full_dofs,) such that D @ k = c.
646        """
647        # Step 1: Perform sparse QR factorization of D^T
648        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
649        Q, R, perm, rank = self._qr(self.lhs.T, economy=False)
650        Q = Q.tocsc()
651        R = R.tocsc()
652
653        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
654        # These correspond to the last n - rank columns of Q
655        C = Q[:, rank:]
656
657        # Step 3: Compute a particular solution k such that D @ k = c
658        # (c = self.rhs contains the prescribed values)
659
660        # 3.1: Apply the column permutation to the RHS vector
661        c_tilde = self.rhs[np.array(perm)]
662
663        # 3.2: Extract the leading square block R1 of R (size rank  rank)
664        R1 = R[:rank, :rank].tocsc()
665
666        # 3.3: Take the first 'rank' entries of the permuted RHS
667        c1 = c_tilde[:rank]
668
669        # 3.4: Solve the triangular system R1^T @ y1 = c1
670        y1 = sps.linalg.spsolve(R1.T, c1)
671
672        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
673        y = np.zeros(Q.shape[0])
674        y[:rank] = y1
675
676        # 3.6: Recover the particular solution k = Q @ y
677        k = Q @ y
678
679        # Step 4: Return the nullspace basis C and the particular solution k
680        return C, k

626        """
627        Construct the transformation (C, k) that enforces the Dirichlet constraints.
628
629        Solves the linear constraint system D @ u = c by computing:
630        - a basis `C` for the nullspace of D (i.e., D @ C = 0),
631        - a particular solution `k` such that D @ k = c.
632
633        Any admissible vector u satisfying the constraints can then be written as:
634
635            u = C @ dof + k
636
637        where `dof` is a reduced set of unconstrained degrees of freedom.
638
639        Returns
640        -------
641        C : sps.spmatrix
642            Basis of the nullspace of D, of shape (n_full_dofs, n_free_dofs), such that D @ C = 0.
643
644        k : np.ndarray[np.floating]
645            Particular solution of shape (n_full_dofs,) such that D @ k = c.
646        """
647        # Step 1: Perform sparse QR factorization of D^T
648        # D^T = Q @ R @ P^T, where D = self.lhs and P is a column permutation
649        Q, R, perm, rank = self._qr(self.lhs.T, economy=False)
650        Q = Q.tocsc()
651        R = R.tocsc()
652
653        # Step 2: Extract a basis C of the nullspace of D (i.e., such that D @ C = 0)
654        # These correspond to the last n - rank columns of Q
655        C = Q[:, rank:]
656
657        # Step 3: Compute a particular solution k such that D @ k = c
658        # (c = self.rhs contains the prescribed values)
659
660        # 3.1: Apply the column permutation to the RHS vector
661        c_tilde = self.rhs[np.array(perm)]
662
663        # 3.2: Extract the leading square block R1 of R (size rank  rank)
664        R1 = R[:rank, :rank].tocsc()
665
666        # 3.3: Take the first 'rank' entries of the permuted RHS
667        c1 = c_tilde[:rank]
668
669        # 3.4: Solve the triangular system R1^T @ y1 = c1
670        y1 = sps.linalg.spsolve(R1.T, c1)
671
672        # 3.5: Complete the vector y by padding with zeros (for the nullspace part)
673        y = np.zeros(Q.shape[0])
674        y[:rank] = y1
675
676        # 3.6: Recover the particular solution k = Q @ y
677        k = Q @ y
678
679        # Step 4: Return the nullspace basis C and the particular solution k
680        return C, k
